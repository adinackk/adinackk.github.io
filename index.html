function getTelegramApi($method, $options = null){

    $str_request = API_URL . TOKEN . '/' . $method;

    if($options){
        $str_request .= '?' . http_build_query($options);
    }
    $request = file_get_contents($str_request);

    return json_decode($request, 1);
}
